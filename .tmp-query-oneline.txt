"/*\n  Multi-warehouse fulfillment: shipment lots + tracking events + aggregated order-facing status.\n\n  - Webhooks update order_shipments (and append order_shipment_tracking_events), not orders.status directly.\n  - recompute_order_fulfillment_aggregate() derives customer_status / order_status / orders.status from all lots.\n*/\n\n-- ─── 1. orders: fulfillment mode ────────────────────────────────────────────────\n\nALTER TABLE public.orders\n  ADD COLUMN IF NOT EXISTS fulfillment_mode text;\n\nCOMMENT ON COLUMN public.orders.fulfillment_mode IS\n  'legacy_single: one Velocity forward order tied to orders row; multi_shipment: use order_shipments lots.';\n\nDO $$ BEGIN\n  ALTER TABLE public.orders\n    ADD CONSTRAINT orders_fulfillment_mode_check\n    CHECK (fulfillment_mode IS NULL OR fulfillment_mode IN ('legacy_single', 'multi_shipment'));\nEXCEPTION WHEN duplicate_object THEN NULL;\nEND $$;\n\n-- ─── 2. Shipment lots (one row per Velocity forward-order / warehouse lot) ───\n\nCREATE TABLE IF NOT EXISTS public.order_shipments (\n  id                    uuid PRIMARY KEY DEFAULT gen_random_uuid(),\n  order_id              uuid NOT NULL REFERENCES public.orders(id) ON DELETE CASCADE,\n  warehouse_id          uuid REFERENCES public.warehouses(id) ON DELETE SET NULL,\n  lot_index             int NOT NULL DEFAULT 1,\n  label                 text NOT NULL DEFAULT 'Shipment 1',\n  velocity_external_code text NOT NULL,\n  velocity_shipment_id   text,\n  velocity_pending_shipment_id text,\n  carrier_shipment_status text,\n  tracking_number        text,\n  velocity_awb           text,\n  velocity_tracking_url  text,\n  velocity_carrier_name  text,\n  velocity_label_url     text,\n  velocity_tracking_snapshot jsonb,\n  created_at             timestamptz NOT NULL DEFAULT now(),\n  updated_at             timestamptz NOT NULL DEFAULT now(),\n  CONSTRAINT order_shipments_order_lot_unique UNIQUE (order_id, lot_index),\n  CONSTRAINT order_shipments_velocity_code_unique UNIQUE (velocity_external_code)\n);\n\nCREATE INDEX IF NOT EXISTS idx_order_shipments_order_id ON public.order_shipments (order_id);\n\n-- ─── 3. Granular tracking events (webhook-driven) ─────────────────────────────\n\nCREATE TABLE IF NOT EXISTS public.order_shipment_tracking_events (\n  id                   uuid PRIMARY KEY DEFAULT gen_random_uuid(),\n  order_shipment_id    uuid NOT NULL REFERENCES public.order_shipments(id) ON DELETE CASCADE,\n  source               text NOT NULL DEFAULT 'webhook',\n  activity             text,\n  location             text,\n  carrier_remark       text,\n  raw_payload          jsonb NOT NULL DEFAULT '{}'::jsonb,\n  event_time           timestamptz,\n  created_at           timestamptz NOT NULL DEFAULT now()\n);\n\nCREATE INDEX IF NOT EXISTS idx_os_te_shipment_time\n  ON public.order_shipment_tracking_events (order_shipment_id, event_time DESC NULLS LAST);\n\n-- ─── 4. Link line items to a lot (optional until admin prepares multi lots) ────\n\nALTER TABLE public.order_items\n  ADD COLUMN IF NOT EXISTS order_shipment_id uuid REFERENCES public.order_shipments(id) ON DELETE SET NULL;\n\nCREATE INDEX IF NOT EXISTS idx_order_items_order_shipment_id\n  ON public.order_items (order_shipment_id)\n  WHERE order_shipment_id IS NOT NULL;\n\n-- ─── 5. Hatvoni order code (matches Edge baseVelocityOrderCode) ─────────────────\n\nCREATE OR REPLACE FUNCTION public.hatvoni_velocity_order_code(p_order_id uuid)\nRETURNS text\nLANGUAGE sql\nIMMUTABLE\nSET search_path = ''\nAS $$\n  SELECT 'HAT-' || upper(substring(replace(p_order_id::text, '-', ''), 1, 10));\n$$;\n\nCREATE OR REPLACE FUNCTION public.hatvoni_velocity_shipment_code(p_order_id uuid, p_lot_index int)\nRETURNS text\nLANGUAGE sql\nIMMUTABLE\nSET search_path = ''\nAS $$\n  SELECT public.hatvoni_velocity_order_code(p_order_id) || '-L' || p_lot_index::text;\n$$;\n\nCREATE OR REPLACE FUNCTION public.resolve_order_from_velocity_external_id(p_ext text)\nRETURNS uuid\nLANGUAGE sql\nSTABLE\nSET search_path = public\nAS $$\n  SELECT o.id\n  FROM public.orders o\n  WHERE o.id::text = trim(p_ext)\n     OR public.hatvoni_velocity_order_code(o.id) = trim(p_ext)\n  LIMIT 1;\n$$;\n\nREVOKE ALL ON FUNCTION public.resolve_order_from_velocity_external_id(text) FROM PUBLIC;\nGRANT EXECUTE ON FUNCTION public.resolve_order_from_velocity_external_id(text) TO service_role;\n\n-- ─── 6. Normalize carrier status → lifecycle bucket ────────────────────────────\n\nCREATE OR REPLACE FUNCTION public.hatvoni_shipment_lifecycle_bucket(p_status text)\nRETURNS text\nLANGUAGE plpgsql\nIMMUTABLE\nSET search_path = ''\nAS $$\nDECLARE\n  s text := lower(trim(coalesce(p_status, '')));\nBEGIN\n  IF s = '' THEN\n    RETURN 'pending_ship';\n  END IF;\n  IF s = 'delivered' THEN\n    RETURN 'delivered';\n  END IF;\n  IF s IN ('cancelled', 'rejected', 'lost') THEN\n    RETURN 'failed';\n  END IF;\n  IF s IN (\n    'pending', 'processing', 'ready_for_pickup', 'pickup_scheduled', 'not_picked'\n  ) THEN\n    RETURN 'pending_ship';\n  END IF;\n  IF s IN (\n    'in_transit', 'out_for_delivery', 'reattempt_delivery', 'externally_fulfilled',\n    'need_attention', 'ndr_raised', 'rto_initiated', 'rto_in_transit', 'rto_need_attention', 'rto_cancelled'\n  ) THEN\n    RETURN 'in_transit';\n  END IF;\n  IF s IN ('rto_delivered') THEN\n    RETURN 'failed';\n  END IF;\n  RETURN 'in_transit';\nEND;\n$$;\n\n-- ─── 7. Aggregate fulfillment state → order columns ─────────────────────────────\n\nCREATE OR REPLACE FUNCTION public.recompute_order_fulfillment_aggregate(p_order_id uuid)\nRETURNS void\nLANGUAGE plpgsql\nSECURITY DEFINER\nSET search_path = public\nAS $$\nDECLARE\n  v_order             public.orders%ROWTYPE;\n  v_mode              text;\n  v_cnt               int;\n  v_del               int;\n  v_fail              int;\n  v_pend              int;\n  v_trans             int;\n  v_customer          text;\n  v_ord_stat          text;\n  v_new_status        public.order_status;\n  v_old_status        public.order_status;\n  v_shipment_roll     text;\nBEGIN\n  SELECT * INTO v_order FROM public.orders WHERE id = p_order_id FOR UPDATE;\n  IF NOT FOUND THEN\n    RETURN;\n  END IF;\n\n  v_mode := v_order.fulfillment_mode;\n\n  SELECT count(*)::int INTO v_cnt FROM public.order_shipments WHERE order_id = p_order_id;\n  IF v_cnt = 0 OR coalesce(v_mode, 'legacy_single') <> 'multi_shipment' THEN\n    RETURN;\n  END IF;\n\n  SELECT\n    count(*) FILTER (WHERE public.hatvoni_shipment_lifecycle_bucket(carrier_shipment_status) = 'delivered'),\n    count(*) FILTER (WHERE public.hatvoni_shipment_lifecycle_bucket(carrier_shipment_status) = 'failed'),\n    count(*) FILTER (WHERE public.hatvoni_shipment_lifecycle_bucket(carrier_shipment_status) = 'pending_ship'),\n    count(*) FILTER (WHERE public.hatvoni_shipment_lifecycle_bucket(carrier_shipment_status) = 'in_transit')\n  INTO v_del, v_fail, v_pend, v_trans\n  FROM public.order_shipments\n  WHERE order_id = p_order_id;\n\n  -- Derive customer-facing strings (prefer granular partial states)\n  IF v_fail > 0 AND v_del + v_trans + v_pend = 0 THEN\n    v_customer := 'cancelled';\n    v_ord_stat := 'cancelled';\n    v_new_status := 'cancelled'::public.order_status;\n  ELSIF v_del = v_cnt THEN\n    v_customer := 'delivered';\n    v_ord_stat := 'delivered';\n    v_new_status := 'delivered'::public.order_status;\n  ELSIF v_del > 0 AND (v_trans > 0 OR v_pend > 0) THEN\n    v_customer := 'partially_delivered';\n    v_ord_stat := 'partially_delivered';\n    v_new_status := v_order.status;\n  ELSIF v_pend > 0 AND (v_trans > 0 OR v_del > 0) THEN\n    v_customer := 'partially_shipped';\n    v_ord_stat := 'partially_shipped';\n    v_new_status := 'shipped'::public.order_status;\n  ELSIF v_trans > 0 AND v_pend = 0 AND v_del = 0 THEN\n    v_customer := 'shipped';\n    v_ord_stat := 'shipped';\n    v_new_status := 'shipped'::public.order_status;\n  ELSIF v_pend = v_cnt THEN\n    v_customer := 'processing';\n    v_ord_stat := 'processing';\n    v_new_status := v_order.status;\n  ELSE\n    v_customer := 'shipped';\n    v_ord_stat := 'shipped';\n    v_new_status := 'shipped'::public.order_status;\n  END IF;\n\n  SELECT string_agg(carrier_shipment_status, ', ' ORDER BY lot_index)\n  INTO v_shipment_roll\n  FROM public.order_shipments\n  WHERE order_id = p_order_id;\n\n  v_old_status := v_order.status;\n\n  PERFORM set_config('hatvoni.workflow_actor', 'system:fulfillment_aggregate', true);\n\n  UPDATE public.orders\n  SET\n    customer_status   = v_customer,\n    order_status      = coalesce(v_ord_stat, order_status),\n    shipment_status   = coalesce(nullif(trim(v_shipment_roll), ''), shipment_status),\n    status            = CASE\n      WHEN v_new_status IS NOT NULL AND v_new_status <> v_old_status THEN v_new_status\n      ELSE status\n    END,\n    shipped_at        = CASE\n      WHEN v_new_status = 'shipped'::public.order_status AND shipped_at IS NULL THEN now()\n      ELSE shipped_at\n    END,\n    processed_at      = CASE\n      WHEN v_new_status = 'delivered'::public.order_status AND processed_at IS NULL THEN now()\n      ELSE processed_at\n    END,\n    updated_at        = now()\n  WHERE id = p_order_id;\n\n  PERFORM set_config('hatvoni.workflow_actor', '', true);\nEND;\n$$;\n\nREVOKE ALL ON FUNCTION public.recompute_order_fulfillment_aggregate(uuid) FROM PUBLIC;\nGRANT EXECUTE ON FUNCTION public.recompute_order_fulfillment_aggregate(uuid) TO service_role;\n\nCOMMENT ON FUNCTION public.recompute_order_fulfillment_aggregate IS\n  'Derives orders.customer_status / status from all order_shipments. Invoked after shipment-level webhook writes.';\n\n-- ─── 8. Admin: create lots from default product warehouses ─────────────────────\n\nCREATE OR REPLACE FUNCTION public.admin_prepare_multi_shipment_lots(p_order_id uuid)\nRETURNS int\nLANGUAGE plpgsql\nSECURITY DEFINER\nSET search_path = public\nAS $$\nDECLARE\n  v_admin uuid;\n  v_order public.orders%ROWTYPE;\n  v_wh    uuid;\n  rec     RECORD;\n  v_idx   int := 0;\n  v_n     int := 0;\nBEGIN\n  SELECT id INTO v_admin FROM public.profiles WHERE id = auth.uid() AND is_admin = true;\n  IF NOT FOUND THEN\n    RAISE EXCEPTION 'Only admins can prepare shipment lots';\n  END IF;\n\n  SELECT * INTO v_order FROM public.orders WHERE id = p_order_id FOR UPDATE;\n  IF NOT FOUND THEN\n    RAISE EXCEPTION 'Order not found';\n  END IF;\n\n  IF v_order.status::text <> 'processing' THEN\n    RAISE EXCEPTION 'Order must be in processing to split shipments';\n  END IF;\n\n  IF EXISTS (SELECT 1 FROM public.order_shipments WHERE order_id = p_order_id) THEN\n    RAISE EXCEPTION 'Shipment lots already exist for this order';\n  END IF;\n\n  IF EXISTS (\n    SELECT 1 FROM public.order_items oi\n    WHERE oi.order_id = p_order_id AND oi.order_shipment_id IS NOT NULL\n  ) THEN\n    RAISE EXCEPTION 'Order items already linked to shipment rows';\n  END IF;\n\n  -- Build distinct warehouses from line items (default warehouse per product / lot source product)\n  FOR rec IN\n    WITH lines AS (\n      SELECT oi.id AS order_item_id,\n        coalesce(\n          pw_direct.warehouse_id,\n          pw_lot.warehouse_id\n        ) AS warehouse_id\n      FROM public.order_items oi\n      LEFT JOIN public.product_warehouses pw_direct\n        ON pw_direct.product_id = oi.product_id AND pw_direct.is_default = true\n      LEFT JOIN public.lots l ON l.id = oi.lot_id\n      LEFT JOIN public.product_warehouses pw_lot\n        ON pw_lot.product_id = l.source_product_id AND pw_lot.is_default = true\n      WHERE oi.order_id = p_order_id\n    )\n    SELECT DISTINCT warehouse_id AS wid FROM lines WHERE warehouse_id IS NOT NULL ORDER BY wid\n  LOOP\n    v_wh := rec.wid;\n    v_idx := v_idx + 1;\n    INSERT INTO public.order_shipments (\n      order_id, warehouse_id, lot_index, label, velocity_external_code\n    ) VALUES (\n      p_order_id,\n      v_wh,\n      v_idx,\n      'Shipment ' || v_idx,\n      public.hatvoni_velocity_shipment_code(p_order_id, v_idx)\n    );\n\n    UPDATE public.order_items oi\n    SET order_shipment_id = (\n      SELECT s.id FROM public.order_shipments s\n      WHERE s.order_id = p_order_id AND s.lot_index = v_idx\n      LIMIT 1\n    )\n    WHERE oi.order_id = p_order_id\n      AND coalesce(\n        (SELECT pw.warehouse_id FROM public.product_warehouses pw\n         WHERE pw.product_id = oi.product_id AND pw.is_default = true LIMIT 1),\n        (SELECT pw2.warehouse_id FROM public.lots l\n         JOIN public.product_warehouses pw2 ON pw2.product_id = l.source_product_id AND pw2.is_default = true\n         WHERE l.id = oi.lot_id LIMIT 1)\n      ) = v_wh;\n\n    v_n := v_n + 1;\n  END LOOP;\n\n  IF v_n = 0 THEN\n    RAISE EXCEPTION 'No warehouse assignments found for order items — set default warehouses on products first';\n  END IF;\n\n  UPDATE public.orders\n  SET fulfillment_mode = 'multi_shipment', updated_at = now(), admin_updated_at = now()\n  WHERE id = p_order_id;\n\n  RETURN v_n;\nEND;\n$$;\n\nREVOKE ALL ON FUNCTION public.admin_prepare_multi_shipment_lots(uuid) FROM PUBLIC;\nGRANT EXECUTE ON FUNCTION public.admin_prepare_multi_shipment_lots(uuid) TO authenticated;\n\n-- ─── 9. RLS ────────────────────────────────────────────────────────────────────\n\nALTER TABLE public.order_shipments ENABLE ROW LEVEL SECURITY;\nALTER TABLE public.order_shipment_tracking_events ENABLE ROW
